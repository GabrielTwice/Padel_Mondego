CREATE OR REPLACE FUNCTION copiar_reserva_para_historico() 
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO historico_reservas (cliente_usuario_id_usuario, id_reserva, data, id_horario, id_camp)
    VALUES (OLD.cliente_usuario_id_usuario, OLD.id_reserva, OLD.data, OLD.id_horario, OLD.id_camp);
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;
